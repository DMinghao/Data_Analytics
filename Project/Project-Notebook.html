mtry.list <- c()
data.forest.final <- NULL
i <- 1
max <- 0
for (i in 1:ncol(data) - 1) {
  data.forest.dummy <- randomForest(
    weight_category ~ .,
    data = data,
    subset = train,
    ntree = 5000,
    mtry = i,
    importance = T
  )
  val <- predict(data.forest.dummy, test, typr = "class")
  score <- mean(val == test$weight_category)
  if (score >= max) {
    data.forest.final <- data.forest.dummy
    max <- score
  }
  mtry.list[i] <- score
}
